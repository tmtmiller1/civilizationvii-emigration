\setlength{\emergencystretch}{4em}

% Better line breaks for code and links.
\usepackage{fvextra}
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{breaklines,breakanywhere,commandchars=\\\{\}}
\IfFileExists{xurl.sty}{\usepackage{xurl}}{}

% Tower-style table readability defaults: denser cells, top alignment,
% gentle row spacing, and full-width longtables.
\usepackage{booktabs}
\usepackage{array}
\usepackage{longtable}
\setlength{\tabcolsep}{4pt}
\renewcommand{\arraystretch}{1.15}
\setlength{\LTleft}{0pt}
\setlength{\LTright}{0pt}

% Apply table style consistently to pandoc-emitted tables.
\AtBeginEnvironment{longtable}{\small}
\AtBeginEnvironment{tabular}{\small}

% The "what counts" matrix uses ✓/✗, which the Latin Modern body font lacks. Map those codepoints
% to pifont dingbats so they render in the PDF (GitHub + in-game render the Unicode directly).
\usepackage{pifont}
\usepackage{newunicodechar}
\newunicodechar{✓}{\ding{51}}
\newunicodechar{✗}{\ding{55}}

% Title and TOC share the first page (no page break between them); the body
% starts on a fresh page after the TOC.
\makeatletter
\let\origmaketitle\maketitle
\renewcommand{\maketitle}{%
  \origmaketitle
  \thispagestyle{empty}
}

\let\origtableofcontents\tableofcontents
\renewcommand{\tableofcontents}{%
  \origtableofcontents
  \clearpage
}
\makeatother
